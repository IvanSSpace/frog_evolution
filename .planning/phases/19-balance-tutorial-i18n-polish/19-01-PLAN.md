    import { describe, it, expect, beforeEach, vi } from 'vitest'
    import { createCosmicSlice, type CosmicState } from './slice'
    import type { CosmicSlice } from './types'
    import type { BoxData } from './types'

    // Minimal Zustand-like harness без полного store.
    function makeStore() {
      let state: CosmicState
      const set = (partial: Partial<CosmicState>) => {
        state = { ...state, ...partial } as CosmicState
      }
      const get = () => state
      state = createCosmicSlice(set, get)
      return { get: () => state, set, getState: () => state }
    }

    function makeBox(overrides: Partial<BoxData> = {}): BoxData {
      return {
        id: overrides.id ?? 'box-1',
        planetId: 'home',
        planetName: 'Home',
        archetype: 'swamp',
        element: 'fire',
        opened: false,
        createdAt: 0,
        ...overrides,
      }
    }

    describe('cosmic.openBox: pity wiring (Phase 19-01)', () => {
      let store: ReturnType<typeof makeStore>

      beforeEach(() => {
        store = makeStore()
      })

      it('common roll increments rare/epic/legendary counters; serum awarded', () => {
        // Force common: replace Math.random чтобы вернуть значение → common range
        // weight common=50 / total 100 → r ∈ [0, 0.5) → common
        const rngSpy = vi.spyOn(Math, 'random').mockReturnValue(0.1)

        const box = makeBox({ id: 'b1', element: 'fire' })
        store.set({ boxes: [box] })

        store.getState().openBox('b1')
        const after = store.getState()

        expect(after.boxes[0].opened).toBe(true)
        expect(after.serums.fire.common).toBe(1)
        expect(after.pityCounters.rare).toBe(1)      // common rolled → rare counter +1
        expect(after.pityCounters.epic).toBe(1)
        expect(after.pityCounters.legendary).toBe(1)
        expect(after.hasOpenedAnyBox).toBe(true)

        rngSpy.mockRestore()
      })

      it('legendary roll resets all pity counters', () => {
        // Need legendary rare branch. Easiest: set pity.legendary = 25 → hard guarantee.
        store.set({
          pityCounters: { common: 0, rare: 5, epic: 5, legendary: 25 },
          boxes: [makeBox({ id: 'b2', element: 'ice' })],
        })

        store.getState().openBox('b2')
        const after = store.getState()

        expect(after.serums.ice.legendary).toBe(1)
        expect(after.pityCounters.rare).toBe(0)
        expect(after.pityCounters.epic).toBe(0)
        expect(after.pityCounters.legendary).toBe(0)
      })

      it('rare guarantee: 3 common streak → next roll cannot be common', () => {
        // pity.rare = 3 → rollRarity returns rare/epic/legendary (70/25/5)
        // Любой Math.random → rare/epic/legendary, никогда не common.
        // Берём seed 0.99 → falls в legendary branch (last 5%)
        // Берём seed 0.5 → rare branch
        const rngSpy = vi.spyOn(Math, 'random').mockReturnValue(0.5)

        store.set({
          pityCounters: { common: 0, rare: 3, epic: 0, legendary: 0 },
          boxes: [makeBox({ id: 'b3', element: 'water' })],
        })

        store.getState().openBox('b3')
        const after = store.getState()

        expect(after.serums.water.common).toBe(0)  // НЕ common
        // и rare counter сбросился (т.к. rolled rare+)
        expect(after.pityCounters.rare).toBe(0)

        rngSpy.mockRestore()
      })

      it('epic guarantee: 10 без epic → next roll guaranteed epic+', () => {
        const rngSpy = vi.spyOn(Math, 'random').mockReturnValue(0.5)

        store.set({
          pityCounters: { common: 0, rare: 0, epic: 10, legendary: 0 },
          boxes: [makeBox({ id: 'b4', element: 'plasma' })],
        })

        store.getState().openBox('b4')
        const after = store.getState()

        // rolled должен быть epic или legendary (не common, не rare)
        expect(after.serums.plasma.common).toBe(0)
        expect(after.serums.plasma.rare).toBe(0)
        expect(after.pityCounters.epic).toBe(0)  // reset

        rngSpy.mockRestore()
      })

      it('hard pity 25: guaranteed legendary on next box', () => {
        store.set({
          pityCounters: { common: 0, rare: 0, epic: 0, legendary: 25 },
          boxes: [makeBox({ id: 'b5', element: 'arcane' })],
        })

        store.getState().openBox('b5')
        const after = store.getState()

        expect(after.serums.arcane.legendary).toBe(1)
        expect(after.pityCounters.legendary).toBe(0)
      })

      it('bonusRarity epic floor upgrades common roll → epic+', () => {
        const rngSpy = vi.spyOn(Math, 'random').mockReturnValue(0.1)
        // Без bonus: rolled = common. С bonus epic — должен upgraded.

        store.set({
          boxes: [makeBox({ id: 'b6', element: 'void', bonusRarity: 'epic' })],
        })

        store.getState().openBox('b6')
        const after = store.getState()

        expect(after.serums.void.common).toBe(0)
        expect(after.serums.void.rare).toBe(0)
        // rolled будет epic или legendary
        const epicCount = after.serums.void.epic + after.serums.void.legendary
        expect(epicCount).toBe(1)

        rngSpy.mockRestore()
      })

      it('opening already-opened box is no-op', () => {
        const box = makeBox({ id: 'b7', opened: true })
        store.set({ boxes: [box] })
        const before = store.getState().pityCounters

        store.getState().openBox('b7')
        const after = store.getState().pityCounters

        expect(after).toEqual(before)
      })
    })
