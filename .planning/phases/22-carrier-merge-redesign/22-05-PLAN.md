<objective>
Реализовать cosmic shop с двумя валютами per D-Cosmic Shop Items + D-Currency Model.

6 items:
1. Cosmic box (L7+ frogs ×3) — essence
2. Permanent slot +1 — essence (scaling)
3. Permanent ship speed +5% — essence (scaling)
4. Permanent serum-drop chance +0.5% — essence (scaling)
5. Skip ship cooldown — серум
6. Trade-up: 3 серума → bonus drop — серум

Purpose: дать sink для обеих валют. Без shop'а essence бессмыслен (только цифра), серум
не имеет long-term application помимо apply-on-frog.

Output: shop config, shopSlice actions, CosmicShopTab UI, новая вкладка в CosmicHubModal,
unit-тесты на cost scaling + purchase atomicity, i18n RU/EN/ES.

    3. Создать `shopSlice.ts`:
       ```ts
       export const createShopSlice = (set, get) => ({
         purchaseShopItem: (itemId, opts?) => {
           const item = SHOP_ITEMS[itemId]
           if (!item) return false
           const state = get().cosmic
           const purchasedTimes = state.shopPurchaseCounts[itemId] ?? 0
           const cost = getNextCost(item, purchasedTimes)

           if (item.currency === 'essence') {
             if (state.essence < cost) return false
           } else {
             const src = opts?.sourceElement ?? opts?.skipElement
             if (!src || state.serums[src] < cost) return false
           }

           set((s) => {
             const next = { ...s.cosmic }
             // Cost deduction
             if (item.currency === 'essence') {
               next.essence -= cost
             } else {
               const src = opts!.sourceElement ?? opts!.skipElement
               next.serums = { ...next.serums, [src]: next.serums[src] - cost }
             }
             next.shopPurchaseCounts = { ...next.shopPurchaseCounts, [itemId]: purchasedTimes + 1 }

             // Apply effect
             switch (itemId) {
               case 'slot_plus_one':       next.permaSlotBonus++; break
               case 'ship_speed':          next.permaShipSpeedBonus++; break
               case 'serum_drop_chance':   next.permaSerumDropBonus++; break
               case 'cosmic_box': {
                 // Emit event; consumer добавит 3 frogs L7 в gameStore
                 eventBus.emit('cosmic:cosmic-box-purchased', {})
                 break
               }
               case 'serum_trade_up': {
                 // +1 random element серум
                 const elements: Element[] = Object.keys(next.serums) as Element[]
                 const random = elements[Math.floor(Math.random() * elements.length)]
                 next.serums = { ...next.serums, [random]: next.serums[random] + 1 }
                 break
               }
               case 'skip_ship_cooldown': {
                 if (s.cosmic.ship?.state === 'transit') {
                   next.ship = { state: 'docked', planetId: s.cosmic.ship.toPlanetId }
                 }
                 break
               }
             }
             return { ...s, cosmic: next }
           })
           return true
         },
       })
       ```

       ```tsx
       import { useState } from 'react'
       import { useTranslation } from 'react-i18next'
       import { useStore } from '../../store/gameStore'
       import { SHOP_ITEMS, getNextCost, type ShopItemId } from '../../config/cosmicShop'

       export function CosmicShopTab() {
         const { t } = useTranslation()
         const essence = useStore(s => s.cosmic.essence)
         const serums = useStore(s => s.cosmic.serums)
         const purchaseCounts = useStore(s => s.cosmic.shopPurchaseCounts)
         const purchase = useStore(s => s.cosmic.purchaseShopItem)
         const [tradeUpElement, setTradeUpElement] = useState<string>('fire')
         const [skipShipElement, setSkipShipElement] = useState<string>('fire')

         return (
           <div style={{ padding: 12, color: '#fff' }}>
             <div style={{ marginBottom: 12 }}>
               <strong>{t('shop.currencies')}: </strong>
               💠 {essence} {t('shop.essence')} · 🧪 {Object.values(serums).reduce((a,b)=>a+b,0)} {t('shop.serum_total')}
             </div>

             <div style={{ display: 'grid', gap: 8 }}>
               {Object.values(SHOP_ITEMS).map(item => {
                 const cost = getNextCost(item, purchaseCounts[item.id] ?? 0)
                 const canAfford = item.currency === 'essence'
                   ? essence >= cost
                   : (item.id === 'serum_trade_up' || item.id === 'skip_ship_cooldown')
                     ? serums[tradeUpElement] >= cost  // используем выбранный
                     : false

                 return (
                   <div key={item.id} style={{
                     padding: 10,
                     border: '1px solid #444',
                     borderRadius: 6,
                     background: canAfford ? '#1f2937' : '#0d1117',
                     opacity: canAfford ? 1 : 0.6,
                   }}>
                     <div style={{ fontWeight: 600 }}>{t(`shop.items.${item.id}.title`)}</div>
                     <div style={{ fontSize: 11, opacity: 0.8 }}>
                       {t(`shop.items.${item.id}.desc`)}
                     </div>
                     <div style={{ marginTop: 6 }}>
                       {t('shop.cost')}: {cost} {item.currency === 'essence' ? '💠' : '🧪'}
                       {' '}({t(`shop.currency.${item.currency}`)})
                     </div>
                     {item.id === 'serum_trade_up' && (
                       <select value={tradeUpElement} onChange={e => setTradeUpElement(e.target.value)}>
                         {(Object.keys(serums) as Array<keyof typeof serums>).map(el => (
                           <option key={el} value={el}>{el} ({serums[el]})</option>
                         ))}
                       </select>
                     )}
                     {item.id === 'skip_ship_cooldown' && (
                       <select value={skipShipElement} onChange={e => setSkipShipElement(e.target.value)}>
                         {(Object.keys(serums) as Array<keyof typeof serums>).map(el => (
                           <option key={el} value={el}>{el} ({serums[el]})</option>
                         ))}
                       </select>
                     )}
                     <button
                       type="button"
                       disabled={!canAfford}
                       onClick={() => {
                         const opts =
                           item.id === 'serum_trade_up' ? { sourceElement: tradeUpElement } :
                           item.id === 'skip_ship_cooldown' ? { skipElement: skipShipElement } :
                           undefined
                         purchase(item.id, opts)
                       }}
                       style={{
                         marginTop: 6, padding: '4px 10px',
                         cursor: canAfford ? 'pointer' : 'not-allowed',
                       }}
                     >
                       {t('shop.buy')}
                     </button>
                   </div>
                 )
               })}
             </div>
           </div>
         )
       }
       ```
