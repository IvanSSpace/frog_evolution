<objective>
Реализовать ascension trigger при достижении L18 carrier'ом + archetype bonus pool.

Это сердце редизайна:
1. Detection: при merge carrier'а до L18 — НЕ требовать дополнительного merge, instant ascend.
2. Ascension effect: carrier исчезает с поля (tween scale/fade + aura pulse), слот освобождается.
3. Permanent archetype bonus добавляется в global pool (4 категории + Прочее).
4. One-shot essence reward (placeholder 1 essence; balance в Plan 22-07).

Purpose: дать carrier'у mechanical payoff. Без ascension carrier забивает слот и ничего не даёт —
это ключевая боль текущего дизайна.

Output: 3 новых артефакта (ascensionSlice, archetypeBonuses, CarrierAscensionTween),
hook в MergeController, unit-тесты на mapping/aggregation/trigger.

       ```ts
       export type ArchetypeCategory = 'fire' | 'water' | 'stone' | 'shadow' | 'other'

       export const ELEMENT_TO_CATEGORY: Record<Element, ArchetypeCategory> = {
         fire: 'fire', plasma: 'fire', war: 'fire',
         water: 'water', forest: 'water', gas: 'water',
         crystal: 'stone', mechanical: 'stone', ring: 'stone',
         shadow: 'shadow', void: 'shadow', arcane: 'shadow', binary: 'shadow',
         ice: 'other', toxic: 'other', desert: 'other',
       }

       export const BONUS_PER_CATEGORY = {
         fire:   { key: 'boxDropSpeed', amount: 0.05 },
         water:  { key: 'tractorGold',  amount: 0.05 },
         stone:  { key: 'offlineCap',   amount: 0.10 },
         shadow: { key: 'serumDrop',    amount: 0.01 },
         other:  { key: 'flatGold',     amount: 0.03 },
       } as const

       // Mini = 10% of full
       export const MINI_BONUS_PER_CATEGORY = {
         fire:   { key: 'boxDropSpeed', amount: 0.005 },
         water:  { key: 'tractorGold',  amount: 0.005 },
         stone:  { key: 'offlineCap',   amount: 0.01  },
         shadow: { key: 'serumDrop',    amount: 0.001 },
         other:  { key: 'flatGold',     amount: 0.003 },
       } as const

       export interface AggregatedBonuses {
         boxDropSpeed: number; tractorGold: number;
         offlineCap: number; serumDrop: number; flatGold: number;
       }

       function emptyBonuses(): AggregatedBonuses {
         return { boxDropSpeed: 0, tractorGold: 0, offlineCap: 0, serumDrop: 0, flatGold: 0 }
       }

       // Full bonus — linear stack per ascended carrier
       export function aggregateFullBonuses(ascended: { element: Element }[]): AggregatedBonuses {
         const result = emptyBonuses()
         for (const c of ascended) {
           const cat = ELEMENT_TO_CATEGORY[c.element]
           const bonus = BONUS_PER_CATEGORY[cat]
           result[bonus.key] += bonus.amount
         }
         return result
       }

       // Mini bonus — max per category (not stacked within same category)
       export function aggregateMiniBonuses(onField: { element: Element }[]): AggregatedBonuses {
         const result = emptyBonuses()
         const categoriesPresent = new Set<ArchetypeCategory>()
         for (const c of onField) {
           categoriesPresent.add(ELEMENT_TO_CATEGORY[c.element])
         }
         for (const cat of categoriesPresent) {
           const bonus = MINI_BONUS_PER_CATEGORY[cat]
           result[bonus.key] += bonus.amount  // only once per category
         }
         return result
       }
       ```

       ```ts
       export const createAscensionSlice = (set, get) => ({
         ascendCarrier: (frogId) => set((state) => {
           const carrier = state.carriers.find(c => c.frogId === frogId)
           if (!carrier) return state
           const ascended: AscendedCarrier = {
             id: `asc-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`,
             element: carrier.element,
             ascendedAt: Date.now(),
           }
           // Emit event для tween (Task 3)
           eventBus.emit('cosmic:carrier-ascended', { frogId, element: carrier.element })
           return {
             ...state,
             carriers: state.carriers.filter(c => c.frogId !== frogId),
             ascendedCarriers: [...state.ascendedCarriers, ascended],
             essence: state.essence + 1,  // placeholder
           }
         }),
       })
       ```
