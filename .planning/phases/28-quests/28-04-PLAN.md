      if (target.kind === 'merge_to_level') return target.level
      if (target.kind === 'raise_relationship') return target.tier
      return target.value

      if (n >= 1e9) return `${(n / 1e9).toFixed(1)}B`
      if (n >= 1e6) return `${(n / 1e6).toFixed(1)}M`
      if (n >= 1e3) return `${(n / 1e3).toFixed(1)}K`
      return String(n)

        case 'relationship_and_bonus': return `${REWARD_ICON.relationship_and_bonus} +1 ${t(RACES_BY_ID[reward.raceId].nameKey)}`
      }

            }} />
          </div>
          <div style={{ fontSize: 10, color: TEXT_VERY_DIM, marginBottom: 10, textAlign: 'right' }}>
            {quest.progress} / {targetVal}
          </div>

    import { SECTION_HEADER_STYLE, TEXT_DIM, TEXT_VERY_DIM, GOLD } from '../_styles'

    const MAX_VISIBLE = 20
    const REWARD_ICON: Record<QuestReward['kind'], string> = {
      essence: '✨', serum: '💉', gold: '🪙', relationship_and_bonus: '🤝',

    function formatGoldShort(n: number): string {

      if (n >= 1e6) return `${(n / 1e6).toFixed(1)}M`
