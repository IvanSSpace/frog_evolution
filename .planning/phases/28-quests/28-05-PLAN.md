    ```tsx
    // Patterns reused from Phase 26-05 FirstContactModal (createPortal + backdrop + Escape).
    // CSS keyframe slide-in (memory feedback_animations — no Lottie).
    // Cliclability-compliant: type=button + touchAction:manipulation + stopPropagation.

    import { useEffect } from 'react'
    import { createPortal } from 'react-dom'
    import { useTranslation } from 'react-i18next'
    import { QUESTS } from '../../game/config/quests'
    import type { QuestReward } from '../../game/config/quests'
    import { RACES_BY_ID } from '../../game/config/races'
    import type { RaceId } from '../../game/config/races'
    import { DARK_CARD_STYLE, PINK_CTA_STYLE, GOLD, TEXT_DIM } from '../CosmicHub/_styles'

    const AUTO_DISMISS_MS = 5000

    const REWARD_ICON: Record<QuestReward['kind'], string> = {
      essence: '✨', serum: '💉', gold: '🪙', relationship_and_bonus: '🤝',
    }

    function formatGoldShort(n: number): string {
      if (n >= 1e9) return `${(n / 1e9).toFixed(1)}B`
      if (n >= 1e6) return `${(n / 1e6).toFixed(1)}M`
      if (n >= 1e3) return `${(n / 1e3).toFixed(1)}K`
      return String(n)
    }

    function rewardSummary(reward: QuestReward, t: (k: string) => string): string {
      switch (reward.kind) {
        case 'essence': return `${REWARD_ICON.essence} ${reward.value}`
        case 'serum': return `${REWARD_ICON.serum} ${reward.count} ${reward.element === 'random' ? '?' : reward.element}`
        case 'gold': return `${REWARD_ICON.gold} ${formatGoldShort(reward.value)}`
        case 'relationship_and_bonus': return `${REWARD_ICON.relationship_and_bonus} +1 ${t(RACES_BY_ID[reward.raceId].nameKey)}`
      }
    }

    interface QuestRewardPopupProps {
      questId: string
      raceId: string
      reward: QuestReward
      onDismiss: () => void
    }

    export function QuestRewardPopup({ questId, raceId, reward, onDismiss }: QuestRewardPopupProps) {
      const { t } = useTranslation()

      // Auto-dismiss timer
      useEffect(() => {
        const tid = window.setTimeout(onDismiss, AUTO_DISMISS_MS)
        return () => window.clearTimeout(tid)
      }, [onDismiss])

      // Escape key dismiss
      useEffect(() => {
        const onKey = (e: KeyboardEvent) => {
          if (e.key === 'Escape') onDismiss()
        }
        window.addEventListener('keydown', onKey)
        return () => window.removeEventListener('keydown', onKey)
      }, [onDismiss])

      const cfg = QUESTS[questId]
      const race = RACES_BY_ID[raceId as RaceId]
      // Defensive: orphan completion event (config removed) — show minimal fallback
      const shortLabel = cfg ? t(cfg.short_key) : questId

      return createPortal(
        <>
          {/* CSS keyframe — mounted inline; CSSOM caches after first render */}
          <style>{`
            @keyframes quest-reward-slide-in {
              0% { transform: translate(-50%, -50%) scale(0.85); opacity: 0; }
              100% { transform: translate(-50%, -50%) scale(1); opacity: 1; }
            }
          `}</style>

          {/* Backdrop — tap dismisses */}
          <div
            onClick={onDismiss}
            style={{
              position: 'fixed',
              inset: 0,
              background: 'rgba(0,0,0,0.55)',
              zIndex: 199,
              touchAction: 'manipulation',
            }}
          />

          {/* Content card — stopPropagation prevents backdrop dismiss */}
          <div
            onClick={(e) => e.stopPropagation()}
            style={{
              position: 'fixed',
              top: '38%',
              left: '50%',
              transform: 'translate(-50%, -50%)',
              zIndex: 200,
              background: '#1a2e1a',
              border: '2px solid rgba(255,255,255,0.18)',
              borderRadius: 16,
              padding: 24,
              maxWidth: 320,
              width: 'calc(100% - 32px)',
              textAlign: 'center',
              boxShadow: '0 8px 32px rgba(0,0,0,0.5)',
              animation: 'quest-reward-slide-in 250ms ease-out',
              touchAction: 'manipulation',
            }}
          >
            {/* Race emoji + sparkle */}
            <div style={{ fontSize: 56, marginBottom: 8, lineHeight: 1 }}>
              {race?.emojiIcon ?? '✨'}
            </div>

            {/* Title */}
            <div style={{
              fontSize: 20,
              fontWeight: 800,
              color: GOLD,
              marginBottom: 8,
              textShadow: '0 1px 0 rgba(0,0,0,0.4)',
            }}>
              {t('cosmic_hub.quests.reward_popup_title')}
            </div>

            {/* Quest short label */}
            <div style={{ fontSize: 13, color: TEXT_DIM, marginBottom: 12, lineHeight: 1.4 }}>
              {shortLabel}
            </div>

            {/* Reward summary */}
            <div style={{
              ...DARK_CARD_STYLE,
              padding: 12,
              marginBottom: 16,
              fontSize: 18,
              fontWeight: 800,
              color: '#fff',
            }}>
              {rewardSummary(reward, t)}
            </div>

            {/* CTA */}
            <button
              type="button"
              onClick={(e) => {
                e.stopPropagation()
                onDismiss()
              }}
              style={{
                ...PINK_CTA_STYLE,
                width: '100%',
              }}
            >
              {t('cosmic_hub.quests.reward_popup_dismiss')}
            </button>
          </div>
        </>,

      )
    }
    ```

    ```tsx
    // Phase 28 Plan 28-05: App-level controller for quest reward popup.
    // Subscribes to eventBus 'quests:completed' and queues popups for sequential display.
    // Renders QuestRewardPopup for head of queue; auto-pops on dismiss.
    // Mirror Phase 26-05 firstContactController pattern (useEffect + useState + cleanup).

    import { useEffect, useState } from 'react'
    import { eventBus } from '../../store/eventBus'
    import type { QuestReward } from '../../game/config/quests'
    import { QuestRewardPopup } from './QuestRewardPopup'

    interface PopupEntry {
      key: string  // unique = activeQuestId from event
      questId: string
      raceId: string
      reward: QuestReward
    }

    const QUEUE_CAP = 10

    export function QuestRewardController() {
      const [queue, setQueue] = useState<PopupEntry[]>([])

      useEffect(() => {
        const onCompleted = (e: {
          questId: string
          raceId: string
          activeQuestId: string
          reward: QuestReward
        }) => {
          setQueue((q) => {
            if (q.length >= QUEUE_CAP) return q  // defensive — drop overflow silently
            // Deduplicate by activeQuestId in case event fires twice (DEV helpers reissue)
            if (q.some((entry) => entry.key === e.activeQuestId)) return q
            return [
              ...q,
              {
                key: e.activeQuestId,
                questId: e.questId,
                raceId: e.raceId,
                reward: e.reward,
              },
            ]
          })
        }
        eventBus.on('quests:completed', onCompleted)
        return () => {
          eventBus.off('quests:completed', onCompleted)
        }
      }, [])

      if (queue.length === 0) return null

      const head = queue[0]
      return (
        <QuestRewardPopup
          key={head.key}
          questId={head.questId}
          raceId={head.raceId}
          reward={head.reward}
          onDismiss={() => setQueue((q) => q.slice(1))}
        />
      )
    }
    ```
