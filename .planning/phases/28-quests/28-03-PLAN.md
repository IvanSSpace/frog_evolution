```typescript
// questEngine.ts public API:

export interface ActivationInput {
  questId: QuestId
  raceId: RaceId
  currentActiveQuests: readonly ActiveQuest[]
  now: number
}

export interface ActivationOutput {
  // null = cap reached OR unknown questId (engine logs warning in DEV)
  newActiveQuest: ActiveQuest | null
  // true if cap reached (caller emits 'quests:cap-reached' event)
  capReached: boolean
}

export function activateQuestFromHook(input: ActivationInput): ActivationOutput

// ─── progress check ─────────────────────────────────────────────────

export interface ProgressInput {
  activeQuests: readonly ActiveQuest[]
  // current snapshot of game state needed for progress calculation
  serumCounts: Record<Element, number>
  goldAmount: number
  discoveredLevels: readonly number[]
  raceRelationships: Record<RaceId, number>
  // event-driven inputs (one event at a time; null = polling tick)
  event:
    | { kind: 'merge'; level: number }
    | { kind: 'box-opened'; element: Element }
    | { kind: 'planet-select'; planetId: string }
    | { kind: 'ship-arrived'; planetId: string }
    | { kind: 'relationship-delta'; raceId: RaceId; newValue: number }
    | null
}

export interface ProgressDelta {
  activeQuestId: string
  newProgress: number
}

export interface ProgressOutput {
  // per-active-quest progress updates (only quests whose progress changed)
  progressDeltas: readonly ProgressDelta[]
  // quests that reached/exceeded target this tick (caller applies reward + moves to history)
  completedQuestIds: readonly string[]
}

export function checkActiveQuestsProgress(input: ProgressInput): ProgressOutput

// ─── reward application (pure — returns delta payload for slice atomic set) ─

export interface RewardApplicationDelta {
  serumDelta?: Record<Element, number>  // {element: count_to_add}
  goldDelta?: number
  essenceDelta?: number
  relationshipDelta?: { raceId: RaceId; delta: number }
  bonusId?: string  // store on completedQuest.rewardClaimed; slice does NOT apply bonus_id (placeholder for Phase 29)
}

export function applyQuestReward(reward: QuestReward, rng?: () => number): RewardApplicationDelta

// ─── helpers ────────────────────────────────────────────────────────

export function isQuestComplete(quest: ActiveQuest, ctx: ProgressInput): boolean
export function generateActiveQuestUuid(now: number): string  // re-use generateActiveQuestId from config/quests.ts
```
