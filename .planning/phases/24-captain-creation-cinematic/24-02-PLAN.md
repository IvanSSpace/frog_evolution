```typescript
// Phase 24 Plan 24-02 — Captain creation cinematic (Beat 2).
//
// «Cosmic Growing Effect»: радиальные particles + 3 concentric rings + camera zoom.
// Длительность ~3 секунды. Triggered eventBus 'captain:birth-start' (x,y) —
// emit'ит 'captain:birth-effect-complete' когда последняя часть (camera return-zoom)
// заканчивается.
//
// IMPORTANT (memory feedback_frog_container_alpha): НЕ трогаем frog.container.alpha.
// Particles + rings — separate GameObjects, depth 9000 (поверх всего).
// Camera zoom — единственный effect который влияет на frogs visually, но это
// масштабирование camera а не alpha, так что мерцания не будет.
//
// Pattern: reuse ConfettiBurst.ts texture-generation подхода (4x4 white pixel
// генерируется один раз на scene, tint накладывается per-particle).
//
// install(scene) — устанавливает subscription к eventBus, parent scene
// делает .install в create() и обязан вызвать .uninstall в shutdown/destroy
// (предотвращает double-listener при hot-reload или scene restart).

import Phaser from 'phaser'
import { eventBus } from '../../store/eventBus'

const TEXTURE_KEY = 'captain-birth-particle'
const DEPTH = 9000

const PALETTE = [0xfde047, 0xffffff, 0x67e8f9] // golden / white / cosmic blue
const PARTICLE_COUNT = 70 // (60-80 per CONTEXT, выбран центр)
const PARTICLE_LIFESPAN_MS = 2500

const RING_COUNT = 3
const RING_OFFSET_MS = 400
const RING_DURATION_MS = 1500
const RING_START_RADIUS = 20
const RING_END_RADIUS = 200
const RING_STROKE_PX = 4
const RING_COLOR = 0xfde047
const RING_START_ALPHA = 0.8

const CAMERA_ZOOM_TARGET = 1.08
const CAMERA_ZOOM_IN_MS = 1500
const CAMERA_ZOOM_OUT_MS = 800

function ensureTexture(scene: Phaser.Scene): void {
  if (scene.textures.exists(TEXTURE_KEY)) return
  const gfx = scene.add.graphics({ x: 0, y: 0 })
  gfx.fillStyle(0xffffff, 1)
  gfx.fillRect(0, 0, 6, 6)
  gfx.generateTexture(TEXTURE_KEY, 6, 6)
  gfx.destroy()
}

function spawnParticles(scene: Phaser.Scene, x: number, y: number): void {
  ensureTexture(scene)
  const emitter = scene.add.particles(x, y, TEXTURE_KEY, {
    speed: { min: 80, max: 200 },          // per CONTEXT
    angle: { min: 0, max: 360 },           // radial spread
    gravityY: 0,                           // per CONTEXT: gravity 0
    lifespan: PARTICLE_LIFESPAN_MS,
    scale: { start: 0.5, end: 2.0 },       // grow
    alpha: { start: 1, end: 0 },
    tint: PALETTE,
    rotate: { min: 0, max: 360 },
    emitting: false,
  })
  emitter.setDepth(DEPTH)
  emitter.explode(PARTICLE_COUNT, x, y)

  scene.time.delayedCall(PARTICLE_LIFESPAN_MS + 300, () => {
    emitter.destroy()
  })
}

function spawnRings(scene: Phaser.Scene, x: number, y: number): void {
  for (let i = 0; i < RING_COUNT; i++) {
    scene.time.delayedCall(i * RING_OFFSET_MS, () => {
      const gfx = scene.add.graphics({ x, y })
      gfx.setDepth(DEPTH - 1) // чуть ниже particles, чтобы particles overlay rings
      const state = { r: RING_START_RADIUS, a: RING_START_ALPHA }

      const draw = () => {
        gfx.clear()
        gfx.lineStyle(RING_STROKE_PX, RING_COLOR, state.a)
        gfx.strokeCircle(0, 0, state.r)
      }
      draw()

      scene.tweens.add({
        targets: state,
        r: RING_END_RADIUS,
        a: 0,
        duration: RING_DURATION_MS,
        ease: 'Sine.easeOut',
        onUpdate: draw,
        onComplete: () => gfx.destroy(),
      })
    })
  }
}

function playCameraZoom(scene: Phaser.Scene, onAllDone: () => void): void {
  const cam = scene.cameras.main
  // Zoom in to target, then back to 1.0 — return-zoom triggers our completion.
  cam.zoomTo(CAMERA_ZOOM_TARGET, CAMERA_ZOOM_IN_MS, 'Sine.easeInOut')
  scene.time.delayedCall(CAMERA_ZOOM_IN_MS + 100, () => {
    cam.zoomTo(1.0, CAMERA_ZOOM_OUT_MS, 'Sine.easeInOut', false, (_cam, progress) => {
      if (progress >= 1) onAllDone()
    })
  })
}

export class CaptainBirthEffect {
  private static handler: ((p: { x: number; y: number }) => void) | null = null

  /**
   * Subscribe scene to captain:birth-start events. Caller (MainScene.create)
   * obtains exclusive ownership of cinematic playback на данной scene.
   * Idempotent: повторный install переустанавливает handler.
   */
  static install(scene: Phaser.Scene): void {
    if (CaptainBirthEffect.handler) {
      eventBus.off('captain:birth-start', CaptainBirthEffect.handler)
    }
    const handler = (p: { x: number; y: number }) => {
      if (!scene.scene.isActive()) return // scene swapped — skip
      CaptainBirthEffect.play(scene, p.x, p.y)
    }
    CaptainBirthEffect.handler = handler
    eventBus.on('captain:birth-start', handler)
  }

  static uninstall(): void {
    if (CaptainBirthEffect.handler) {
      eventBus.off('captain:birth-start', CaptainBirthEffect.handler)
      CaptainBirthEffect.handler = null
    }
  }

  /**
   * Play full cinematic at (x,y). Fires 'captain:birth-effect-complete' after
   * camera return-zoom finishes (~3 секунды total).
   */
  static play(scene: Phaser.Scene, x: number, y: number): void {
    spawnParticles(scene, x, y)
    spawnRings(scene, x, y)
    playCameraZoom(scene, () => {
      eventBus.emit('captain:birth-effect-complete')
    })
  }
}
```
