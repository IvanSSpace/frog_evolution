    ```ts
    // Phase 26 Plan 26-05 — First contact cinematic.
    //
    // Lighter-scale version of CaptainBirthEffect (Phase 24):
    //   - ~35 particles (vs 70) tinted с race.homeColor + white
    //   - 1 expanding ring (vs 3 cascade) — race color
    //   - No camera zoom (читы pre-existing camera state)
    //   - ~2s total duration (vs 3s)
    //
    // Trigger: eventBus 'cosmos:first-contact' с {raceId, x, y}.
    // On completion: emit 'cosmos:first-contact-effect-complete' (no payload).
    //
    // Mount: installFirstContactEffect(scene) подписывает global handler.
    //         Idempotent через scene.textures.exists для texture-gen.
    //         install on first scene mount (MainScene или StarMapScene — обе работают);
    //         Wave 1 — install после scene.create если scene already running.
    //
    // НЕ trogаем frog.container alpha — все GameObjects independent depth 9000.

    import Phaser from 'phaser'
    import { eventBus } from '../../store/eventBus'
    import { RACES_BY_ID, type RaceId } from '../config/races'

    const TEXTURE_KEY = 'first-contact-particle'
    const DEPTH = 9000
    
    const PARTICLE_COUNT = 35
    const PARTICLE_LIFESPAN_MS = 1800
    const PARTICLE_SPEED_MIN = 60
    const PARTICLE_SPEED_MAX = 160
    
    const RING_DURATION_MS = 1500
    const RING_START_RADIUS = 12
    const RING_END_RADIUS = 140
    const RING_STROKE_PX = 3
    const RING_START_ALPHA = 0.8
    
    const TOTAL_DURATION_MS = 2000  // emit complete после этого

    function ensureTexture(scene: Phaser.Scene): void {
      if (scene.textures.exists(TEXTURE_KEY)) return
      const gfx = scene.add.graphics({ x: 0, y: 0 })
      gfx.fillStyle(0xffffff, 1)
      gfx.fillRect(0, 0, 4, 4)
      gfx.generateTexture(TEXTURE_KEY, 4, 4)
      gfx.destroy()
    }

    function getScene(): Phaser.Scene | null {
      // Helper — first contact may fire when player on Star Map OR Main scene.
      // window.__starMapScene OR window.__mainScene — use whichever is active.
      const w = window as unknown as { __starMapScene?: Phaser.Scene; __mainScene?: Phaser.Scene }
      return w.__starMapScene ?? w.__mainScene ?? null
    }

    function spawnParticles(scene: Phaser.Scene, x: number, y: number, color: number): void {
      ensureTexture(scene)
      const emitter = scene.add.particles(x, y, TEXTURE_KEY, {
        speed: { min: PARTICLE_SPEED_MIN, max: PARTICLE_SPEED_MAX },
        angle: { min: 0, max: 360 },
        lifespan: PARTICLE_LIFESPAN_MS,
        scale: { start: 0.4, end: 1.6 },
        alpha: { start: 1, end: 0 },
        gravityY: 0,
        tint: [color, 0xffffff],
        quantity: PARTICLE_COUNT,
        emitting: false,
      })
      emitter.setDepth(DEPTH)
      emitter.explode(PARTICLE_COUNT, x, y)
      scene.time.delayedCall(PARTICLE_LIFESPAN_MS + 200, () => emitter.destroy())
    }

    function spawnRing(scene: Phaser.Scene, x: number, y: number, color: number): void {
      const ring = scene.add.graphics({ x, y })
      ring.setDepth(DEPTH)
      let radius = RING_START_RADIUS
      let alpha = RING_START_ALPHA
      const draw = () => {
        ring.clear()
        ring.lineStyle(RING_STROKE_PX, color, alpha)
        ring.strokeCircle(0, 0, radius)
      }
      draw()
      scene.tweens.add({
        targets: { r: RING_START_RADIUS, a: RING_START_ALPHA },
        r: RING_END_RADIUS,
        a: 0,
        duration: RING_DURATION_MS,
        ease: 'Quad.easeOut',
        onUpdate: (tween) => {
          radius = tween.getValue('r') ?? RING_END_RADIUS
          alpha = tween.getValue('a') ?? 0
          draw()
        },
        onComplete: () => ring.destroy(),
      })
    }

    let activeHandler: ((p: { raceId: string; x: number; y: number }) => void) | null = null

    export function installFirstContactEffect(): () => void {
      if (activeHandler) {
        eventBus.off('cosmos:first-contact', activeHandler)
        activeHandler = null
      }
      
      const handler = (payload: { raceId: string; x: number; y: number }) => {
        const scene = getScene()
        if (!scene) {
          console.warn('[FirstContactEffect] no active scene — skip cinematic, modal still fires')
          // Still emit complete чтобы modal не залип в pending state.
          scene?.time?.delayedCall?.(0, () => 
            eventBus.emit('cosmos:first-contact-effect-complete')
          ) ?? setTimeout(() => 
            eventBus.emit('cosmos:first-contact-effect-complete'), 0
          )
          return
        }
        const race = RACES_BY_ID[payload.raceId as RaceId]
        if (!race) return
        
        spawnParticles(scene, payload.x, payload.y, race.homeColor)
        spawnRing(scene, payload.x, payload.y, race.homeColor)
        
        scene.time.delayedCall(TOTAL_DURATION_MS, () => {
          eventBus.emit('cosmos:first-contact-effect-complete')
        })
      }
      
      eventBus.on('cosmos:first-contact', handler)
      activeHandler = handler
      
      return () => {
        if (activeHandler === handler) {
          eventBus.off('cosmos:first-contact', handler)
          activeHandler = null
        }
      }
    }

    export function uninstallFirstContactEffect(): void {
      if (activeHandler) {
        eventBus.off('cosmos:first-contact', activeHandler)
        activeHandler = null
      }
    }
    ```

    Подход: пришедший event `'starmap:planet-tapped'` checks inhabitant + firstContactsSeen gate, emit'ит `'cosmos:first-contact'` если первый раз. Phaser handler из Task 1 запускает cinematic. После cinematic 'cosmos:first-contact-effect-complete' → controller mounts FirstContactModal.

    ```tsx
    // Phase 26 Plan 26-05 — First contact coordinator.
    //
    // Lifecycle:
    //   1. StarMapScene emits 'starmap:planet-tapped' {id, type, ...}
    //   2. Controller checks getPlanetInhabitant(id) → if inhabitant AND 
    //      !firstContactsSeen[inhabitant.raceId] → emit 'cosmos:first-contact'
    //      {raceId, x, y} (x,y from StarMapScene planet world coords → screen).
    //   3. FirstContactEffect.ts (Task 1) handles 'cosmos:first-contact':
    //      runs Phaser cinematic ~2s → emit 'cosmos:first-contact-effect-complete'.
    //   4. Controller подписан на 'cosmos:first-contact-effect-complete' → setState 
    //      pending raceId → mount FirstContactModal.
    //   5. Modal CTA/backdrop → markFirstContactSeen + onClose → state cleared.
    //
    // Mount: один раз в App.tsx (рядом с CaptainBirthController / OnboardingController).

    import { useEffect, useState } from 'react'
    import { eventBus } from '../../store/eventBus'
    import { useGameStore } from '../../store/gameStore'
    import { getPlanetInhabitant } from '../../game/data/habitablePlanets'
    import { installFirstContactEffect } from '../../game/effects/FirstContactEffect'
    import { FirstContactModal } from './FirstContactModal'
    import type { RaceId } from '../../game/config/races'

    export function FirstContactController() {
      const [pendingRaceId, setPendingRaceId] = useState<RaceId | null>(null)
      
      useEffect(() => {
        // Install Phaser effect handler (auto-cleanup on unmount).
        const cleanup = installFirstContactEffect()
        
        // Handler 1: planet-tapped → maybe emit first-contact.
        const onPlanetTapped = (p: { id: string; type: string; durationMs: number; seed: number }) => {
          const inhabitant = getPlanetInhabitant(p.id)
          if (!inhabitant) return
          
          const seen = useGameStore.getState().firstContactsSeen[inhabitant.raceId as RaceId]
          if (seen === true) return  // idempotent
          
          // Get planet world coords для cinematic anchor.
          // StarMapScene стreaming planet positions через some API; для MVP — use 
          // window.__starMapScene to query planet coords если scene активна.
          // Fallback: screen center.
          const w = window as unknown as { 
            __starMapScene?: { 
              cameras?: { main?: { centerX: number; centerY: number } }
              getPlanetWorldCoords?: (id: string) => { x: number; y: number } | null
            }
          }
          let x = 200
          let y = 300
          if (w.__starMapScene) {
            const fromHelper = w.__starMapScene.getPlanetWorldCoords?.(p.id)
            if (fromHelper) {
              x = fromHelper.x
              y = fromHelper.y
            } else {
              const cam = w.__starMapScene.cameras?.main
              if (cam) {
                x = cam.centerX
                y = cam.centerY
              }
            }
          }
          
          eventBus.emit('cosmos:first-contact', {
            raceId: inhabitant.raceId,
            x, y,
          })
        }
        
        // Handler 2: effect complete → show DOM modal.
        const onEffectComplete = () => {
          // Read которая raceId «в полёте» — emit события 'cosmos:first-contact'
          // pull payload сложно (mitt не каше'ит last). Подход: capture в outer 
          // closure из emit-handler.
          // Реализуется через outer ref-style:
          if (pendingRaceIdInFlightRef.current) {
            setPendingRaceId(pendingRaceIdInFlightRef.current)
            pendingRaceIdInFlightRef.current = null
          }
        }
        
        // Захватываем raceId для effect-complete (cinematic completion).
        const pendingRaceIdInFlightRef = { current: null as RaceId | null }
        const onFirstContactEmit = (p: { raceId: string; x: number; y: number }) => {
          pendingRaceIdInFlightRef.current = p.raceId as RaceId
        }
        
        eventBus.on('starmap:planet-tapped', onPlanetTapped)
        eventBus.on('cosmos:first-contact', onFirstContactEmit)
        eventBus.on('cosmos:first-contact-effect-complete', onEffectComplete)
        
        return () => {
          eventBus.off('starmap:planet-tapped', onPlanetTapped)
    ```
