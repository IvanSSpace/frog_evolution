    Создать `client/src/game/scenes/starmap/effects/raceGlow.ts`. Этот модуль — единая точка для всех race-overlay visuals (glow + icon + home pulse halo).

    **Архитектура:**
    - `RaceGlowController` класс с `attach(planet)` / `detach(planetId)` / `destroy()` методами.
    - Контроллер хранит `Map<planetId, OverlayGroup>` где `OverlayGroup = { glow: Phaser.GameObjects.Image, icon: Phaser.GameObjects.Text, homeHalo?: Phaser.GameObjects.Image, homeHaloTween?: Phaser.Tweens.Tween }`.
    - Texture-gen pattern (reuse ConfettiBurst.ts pattern): radial gradient white texture сгенерирована единожды per scene (key 'race-glow-radial'), tint накладывается per-overlay.
    - Никакого Lottie (memory `feedback_animations`).
    - НЕ tween'ить alpha frog.container или planet.container — отдельные overlay GameObjects (memory `feedback_frog_container_alpha`).

    **Содержимое:**

    ```ts
    // Phase 26 Plan 26-03: race-overlay visuals для Star Map habitable planets.
    //
    // Glow: radial gradient white texture (generated один раз per scene), tinted
    //       per-race через ELEMENT_TINTS[affinity]. Radius = planet.size + 8.
    //       Alpha 0.5 для colony, 0.65 для home. depth = planet depth - 1 (за planet'ой).
    //
    // Home halo: дополнительный gold (#fde047) thin halo radius=planet.size+18,
    //       alpha 0.4, infinite pulse tween (alpha 0.2 ↔ 0.6, 1500ms, yoyo).
    //
    // Icon: Phaser.Text emoji от race.emojiIcon (depth = planet depth + 1, поверх planet).
    //       Home: size 18px (фронт-bold). Colony: size 14px.
    //
    // Cleanup: detach(id) удаляет 3 game objects + tween. destroy() — все entries.
    //
    // НЕ trogаем planet.container alpha — все overlays отдельные GameObjects.

    import Phaser from 'phaser'
    import { getRaceColor, RACES_BY_ID, type RaceId } from '../../../config/races'
    import type { PlanetInhabitant } from '../../../../store/cosmic/types'

    const GLOW_TEX_KEY = 'race-glow-radial-256'
    const GOLD_HALO_COLOR = 0xfde047
    const HOME_PULSE_DURATION_MS = 1500

    interface OverlayGroup {
      glow: Phaser.GameObjects.Image
      icon: Phaser.GameObjects.Text
      homeHalo?: Phaser.GameObjects.Image
      homeHaloTween?: Phaser.Tweens.Tween
    }

    function ensureGlowTexture(scene: Phaser.Scene): void {
      if (scene.textures.exists(GLOW_TEX_KEY)) return
      const size = 256
      const gfx = scene.add.graphics({ x: 0, y: 0 })
      // Radial gradient через concentric circles.
      const steps = 32
      for (let i = steps; i > 0; i--) {
        const radius = (size / 2) * (i / steps)
        const alpha = (1 - i / steps) ** 2  // sharper falloff
        gfx.fillStyle(0xffffff, alpha)
        gfx.fillCircle(size / 2, size / 2, radius)
      }
      gfx.generateTexture(GLOW_TEX_KEY, size, size)
      gfx.destroy()
    }

    export interface RaceGlowAttachInput {
      planetId: string
      x: number
      y: number
      size: number      // planet visual radius
      depth: number     // planet depth (overlay sits depth-1 / depth+1)
      inhabitant: PlanetInhabitant
    }

    export class RaceGlowController {
      private scene: Phaser.Scene
      private overlays = new Map<string, OverlayGroup>()
      
      constructor(scene: Phaser.Scene) {
        this.scene = scene
        ensureGlowTexture(scene)
      }
      
      attach(input: RaceGlowAttachInput): void {
        if (this.overlays.has(input.planetId)) return // idempotent
        const race = RACES_BY_ID[input.inhabitant.raceId as RaceId]
        if (!race) return // defensive — unknown raceId
        
        const color = getRaceColor(input.inhabitant.raceId as RaceId)
        const isHome = input.inhabitant.role === 'home'
        const glowRadius = input.size + 8
        const glowAlpha = isHome ? 0.65 : 0.5
        
        // Glow halo
        const glow = this.scene.add.image(input.x, input.y, GLOW_TEX_KEY)
          .setDepth(input.depth - 1)
          .setTint(color)
          .setAlpha(glowAlpha)
          .setDisplaySize(glowRadius * 2, glowRadius * 2)
          .setBlendMode(Phaser.BlendModes.ADD)
        
        // Icon overlay (emoji)
        const icon = this.scene.add.text(input.x, input.y, race.emojiIcon, {
          fontSize: isHome ? '18px' : '14px',
          fontStyle: isHome ? 'bold' : 'normal',
        }).setOrigin(0.5).setDepth(input.depth + 1)
        
        // Home: additional gold pulsing halo
        let homeHalo: Phaser.GameObjects.Image | undefined
        let homeHaloTween: Phaser.Tweens.Tween | undefined
        if (isHome) {
          const haloRadius = input.size + 18
          homeHalo = this.scene.add.image(input.x, input.y, GLOW_TEX_KEY)
            .setDepth(input.depth - 2)
            .setTint(GOLD_HALO_COLOR)
            .setAlpha(0.4)
            .setDisplaySize(haloRadius * 2, haloRadius * 2)
            .setBlendMode(Phaser.BlendModes.ADD)
          homeHaloTween = this.scene.tweens.add({
            targets: homeHalo,
            alpha: { from: 0.2, to: 0.6 },
            duration: HOME_PULSE_DURATION_MS,
            yoyo: true,
            repeat: -1,
            ease: 'Sine.easeInOut',
          })
        }
        
        this.overlays.set(input.planetId, { glow, icon, homeHalo, homeHaloTween })
      }
      
      detach(planetId: string): void {
        const o = this.overlays.get(planetId)
        if (!o) return
        o.glow.destroy()
        o.icon.destroy()
        o.homeHalo?.destroy()
        o.homeHaloTween?.stop()
        o.homeHaloTween?.remove()
        this.overlays.delete(planetId)
      }
      
      /** При camera/planet movement (panning, follow) — update positions. */
      updatePosition(planetId: string, x: number, y: number): void {
        const o = this.overlays.get(planetId)
        if (!o) return
        o.glow.setPosition(x, y)
        o.icon.setPosition(x, y)
        o.homeHalo?.setPosition(x, y)
      }
      
      destroy(): void {
        for (const id of Array.from(this.overlays.keys())) {
          this.detach(id)
        }
        this.overlays.clear()
      }
      
      get attachedCount(): number {
        return this.overlays.size
      }
    }
    ```

    **Edge cases handled:**
    - Duplicate `attach()` для того же planetId — idempotent (return).
    - Unknown raceId (corrupt planetMap) — defensive return без error.
    - destroy() очищает tweens (избегаем leak'и при scene shutdown).
