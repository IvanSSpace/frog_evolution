    2. Реализация:
       ```ts
       applySerum: (frogId, element, rarity, level) => {
         const s = get()

         // Guard: serum availability
         const cur = s.serums[element][rarity]
         if (cur < 1) return

         // Guard: idempotency — frog уже carrier?
         if (s.carriers.some((c) => c.frogId === frogId)) return

         const nextSerums = {
           ...s.serums,
           [element]: { ...s.serums[element], [rarity]: cur - 1 },
         }
         const nextCarrier: CarrierData = {
           frogId, element, rarity,
           feedCount: 0,
           stabilized: false,
           level,
         }
         set({
           serums: nextSerums,
           carriers: [...s.carriers, nextCarrier],
           serumDragActive: false,
           selectedSerum: null,
         })
       }
       ```

    **Не использовать существующие `removeSerum`/`addCarrier` через get()/set()
    composition** — atomic single-`set()` гарантирует, что
    `FrogOverlayManager.subscribe` отрабатывает один раз (одно reference change).
    Это критично для Phase 12 manager логики (он использует `state.carriers !== last`).

    ```ts
    // Phase 14: pure eligibility check для Serum-02 → Carrier flow.
    // Locked decisions (REQ SERUM-08):
    //   common → L1, rare → L7, epic → L13, legendary → L19.
    //   Не применима, если frog уже carrier.

    import type { Element, Rarity, CarrierData } from '../store/cosmic/types'

    /** Required frog level per serum rarity (REQ SERUM-08, locked). */
    export const RARITY_TO_STARTING_LEVEL: Record<Rarity, number> = {
      common: 1,
      rare: 7,
      epic: 13,
      legendary: 19,
    }

    /** Minimal frog shape — accepts MainScene FrogData без import cycle. */
    export interface FrogLike {
      id: string
      level: number
    }

    /**
     * Pure check: можно ли применить серум данной rarity на этого frog'а?
     * - level должен совпадать с RARITY_TO_STARTING_LEVEL[rarity]
     * - frog не должен уже быть carrier
     * Element не используется в проверке eligibility (любой element подходит
     * eligible frog'е) — только rarity gates level.
     */
    export function isEligible(
      frog: FrogLike,
      _element: Element,
      rarity: Rarity,
      carriers: ReadonlyArray<CarrierData>,
    ): boolean {
      if (carriers.some((c) => c.frogId === frog.id)) return false
      return frog.level === RARITY_TO_STARTING_LEVEL[rarity]
    }

    /** Helper для UI: required level + location label for mis-tap toast. */
    export function getEligibilityHint(rarity: Rarity): { level: number; locationId: number } {
      const level = RARITY_TO_STARTING_LEVEL[rarity]
      // Локация определяется по level: 1-6 → Болото, 7-12 → Лес, 13-18 → Континент, 19-24 → Планета.
      const locationId =
        level <= 6 ? 1 :
        level <= 12 ? 2 :
        level <= 18 ? 3 : 4
      return { level, locationId }
    }
    ```

    **Поле `_element` с подчёркиванием** — TypeScript-конвенция для unused
    параметров. Сохраняем сигнатуру (element, rarity, carriers) на случай если
    в Phase 17 потребуется element-specific логика (например — двойной apply).

    Тесты НЕ создаём (нет vitest setup в проекте). Behavior проверяется через
    smoke в Plan 14-04.
