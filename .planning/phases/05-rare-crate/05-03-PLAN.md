
```typescript

  // ... existing cleanup ...
  eventBus.off('rareCrate:opened', handleRareCrateOpened)
```

```typescript

```

```tsx

    minLevel={rareCrate.minLevel}
    maxLevel={rareCrate.maxLevel}
    onClose={handleRareCrateClaim}
  />
)}
```
