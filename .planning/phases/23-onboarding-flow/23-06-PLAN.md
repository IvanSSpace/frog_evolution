    ```js
    const ru = require('../client/src/i18n/ru.json')
    const en = require('../client/src/i18n/en.json')

    const requiredOnb = [
      'onboarding.welcome.title',
      'onboarding.welcome.subtitle',
      'onboarding.welcome.cta',
      'onboarding.tapHint.label',
      'onboarding.mergeHint.label',
      'onboarding.mergeHint.success',
      'onboarding.location.unlocked',
    ]

      return path.split('.').reduce((a, k) => a?.[k], obj)

        if (!get(locObj, key)) {
    ```

    ```typescript
    import { useOnboardingStore } from '../store/onboarding/onboardingSlice'
    import { eventBus } from '../store/eventBus'

    export function installOnboardingDevHelpers(): void {
      if (!import.meta.env.DEV) return
      const w = window as any

      w.__resetOnboarding = () => {
        useOnboardingStore.getState().__reset()
        window.location.reload()
      }

      w.__skipOnboarding = () => {
        const s = useOnboardingStore.getState()
        s.markWelcomeSeen()
        s.markFirstBoxTapSeen()
        s.markFirstMergeSeen()
        ;[2, 3, 6].forEach((id) => s.markLocationCelebrated(id))
        console.info('[onboarding] all flags set true')
      }

      // Beat-specific test triggers — позволяют jump'нуть к конкретному beat
      // без проходжения предыдущих.

      /**
       * Trigger Beat 2 manually: emit tutorial:firstBoxSpawned с fake coords.
       * Useful когда не хочется ждать реального box drop.
       * NB: ring не появится без реального бокса — этот helper в основном для DOM label test.
       */
      w.__triggerBeat2 = () => {
        const s = useOnboardingStore.getState()
        if (!s.welcomeSeen) s.markWelcomeSeen()
        // Reset Beat 2 flag чтобы trigger сработал
        useOnboardingStore.setState({ firstBoxTapSeen: false })
        eventBus.emit('tutorial:firstBoxSpawned', {
          x: 200,
          y: 300,
          boxId: 'dev-fake',
          width: 80,
        })
      }

      /**
       * Trigger Beat 4 для конкретной локации.
       * Usage: __triggerBeat4(2) — Болото; __triggerBeat4(3) — Лес; __triggerBeat4(6) — Star Map.
       */
      w.__triggerBeat4 = (locationId: number) => {
        if (![2, 3, 6].includes(locationId)) {
          console.error('[onboarding] __triggerBeat4: locationId must be 2, 3 or 6')
          return
        }
        // Reset celebration flag чтобы trigger сработал
        const s = useOnboardingStore.getState()
        const next = { ...s.locationsCelebrated }
        delete next[locationId]
        useOnboardingStore.setState({ locationsCelebrated: next })
        eventBus.emit('location:unlocked', { locationId })
      }

      /**
       * Inspect: print current onboarding state.
       */
      w.__onboardingState = () => {
        const s = useOnboardingStore.getState()
        const snap = {
          welcomeSeen: s.welcomeSeen,
          firstBoxTapSeen: s.firstBoxTapSeen,
          firstMergeSeen: s.firstMergeSeen,
          locationsCelebrated: s.locationsCelebrated,
        }
        console.table(snap)
        return snap
      }

      console.info('[onboarding] dev helpers installed: __resetOnboarding, __skipOnboarding, __triggerBeat2, __triggerBeat4, __onboardingState')
    }
    ```
