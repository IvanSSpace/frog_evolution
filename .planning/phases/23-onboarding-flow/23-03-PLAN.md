    ```typescript
    import Phaser from 'phaser'

    export interface TutorialPulseRingOptions {
      scene: Phaser.Scene
      // Target: any GameObject с .x/.y/.active. Ring следует за ним.
      target: { x: number; y: number; active: boolean }
      radius: number
      // Default #ec4899
      color?: number
      // Default 3
      strokeWidth?: number
      // Default 800ms
      duration?: number
      // Default 1.15
      maxScale?: number
      // Layer: render выше боксов/лягушек, но ниже UI. Default 5000.
      depth?: number
    }

    /**
     * Reusable Phaser pulsing ring effect.
     * Used in: Beat 2 (box hint, Plan 23-03), Beat 3 (frog hints, Plan 23-04).
     *
     * Lifecycle:
     *   new TutorialPulseRing(opts) → ring появляется и трекает target.
     *   .destroy() → fade-out 300ms потом cleanup.
     */
    export class TutorialPulseRing {
      private container: Phaser.GameObjects.Container
      private tween: Phaser.Tweens.Tween
      private updateHandler: () => void
      private destroyed = false

      constructor(private opts: TutorialPulseRingOptions) {
        const scene = opts.scene
        const color = opts.color ?? 0xec4899
        const stroke = opts.strokeWidth ?? 3
        const duration = opts.duration ?? 800
        const maxScale = opts.maxScale ?? 1.15
        const depth = opts.depth ?? 5000

        const ring = scene.add.graphics()
        ring.lineStyle(stroke, color, 1)
        ring.strokeCircle(0, 0, opts.radius)

        this.container = scene.add.container(opts.target.x, opts.target.y, [ring])
        this.container.setDepth(depth)
        this.container.setAlpha(0.4)
        // pointer-events should not block taps. Container default doesn't have hit area.

        this.tween = scene.tweens.add({
          targets: this.container,
          alpha: { from: 0.4, to: 0.9 },
          scale: { from: 1.0, to: maxScale },
          duration,
          yoyo: true,
          repeat: -1,
          ease: 'Sine.easeInOut',
        })

        // Follow target each frame.
        this.updateHandler = () => {
          if (this.destroyed) return
          if (!opts.target.active) return
          this.container.x = opts.target.x
          this.container.y = opts.target.y
        }
        scene.events.on('update', this.updateHandler)
      }

      // Returns current x/y for DOM-overlay anchor calculation.
      get worldPosition(): { x: number; y: number } {
        return { x: this.container.x, y: this.container.y }
      }

      /**
       * Fade-out 300ms, потом destroy. Idempotent.
       */
      destroy(fadeMs = 300): void {
        if (this.destroyed) return
        this.destroyed = true

        this.tween.stop()
        this.opts.scene.events.off('update', this.updateHandler)

        this.opts.scene.tweens.add({
          targets: this.container,
          alpha: 0,
          scale: 1.2,
          duration: fadeMs,
          ease: 'Sine.easeOut',
          onComplete: () => {
            this.container.destroy()
          },
        })
      }
    }
    ```

    **Important — Cliclability:**
    Container НЕ ставим interactive (`setInteractive`). Без hit area = pointer events
    проходят сквозь ring к боксу под ним. Это сохраняет tap-on-box работающим.

    **Important — Tracking:**
    `scene.events.on('update', ...)` запускается каждый frame. Если box destroy'ится
    (player tap), `target.active` становится false → ring замораживается на месте,
    потом dismiss-handler вызовет `destroy()`.

    **`TapHintOverlay.tsx`:**
    DOM label «Тапни 👆» позиционированный под боксом. Phaser canvas координаты
    конвертируем в DOM via canvas.getBoundingClientRect() + scene scale.

    Простейший вариант — слушаем 'tutorial:firstBoxSpawned' event с {x, y, width}
    в game-coordinates, передаём через ref'ы game scene → DOM offset.

    ```tsx
    import { useEffect, useState } from 'react'
    import { useTranslation } from 'react-i18next'
    import { eventBus } from '../../store/eventBus'
    import { useOnboardingStore } from '../../store/onboarding/onboardingSlice'

    interface SpawnPayload {
      x: number
      y: number
      boxId: string
      width: number
    }

    /**
     * Phase 23 Beat 2 — DOM label «Тапни 👆» под box ring.
     * Auto-fade через 5с или при box tap.
     *
     * Позиционирование: используем game canvas bounding rect.
     * Phaser ring сам по себе работает в game-coords; DOM label позиционируется
     * через `position: fixed` поверх canvas.
     */
    export function TapHintOverlay() {
      const { t } = useTranslation()
      const [spawn, setSpawn] = useState<SpawnPayload | null>(null)
      const [exiting, setExiting] = useState(false)

      useEffect(() => {
        const onSpawn = (p: SpawnPayload) => {
          // Только если state ещё не seen.
          const { welcomeSeen, firstBoxTapSeen } = useOnboardingStore.getState()
          if (!welcomeSeen || firstBoxTapSeen) return
          setSpawn(p)
          setExiting(false)
        }
        const onTapped = () => {
          setExiting(true)
          window.setTimeout(() => setSpawn(null), 300)
        }
        eventBus.on('tutorial:firstBoxSpawned', onSpawn)
        eventBus.on('tutorial:firstBoxTapped', onTapped)
        return () => {
          eventBus.off('tutorial:firstBoxSpawned', onSpawn)
          eventBus.off('tutorial:firstBoxTapped', onTapped)
        }
      }, [])

      // Auto-fade 5с
      useEffect(() => {
        if (!spawn || exiting) return
        const timer = window.setTimeout(() => {
          setExiting(true)
          window.setTimeout(() => {
            setSpawn(null)
            // Помечаем seen (auto-dismiss считается seen)
            useOnboardingStore.getState().markFirstBoxTapSeen()
          }, 300)
        }, 5000)
        return () => clearTimeout(timer)
      }, [spawn, exiting])

      if (!spawn) return null

      // Convert game coords → DOM coords.
      // Phaser canvas — element id `#game-canvas` или first <canvas>.
      const canvas = document.querySelector('canvas')
      if (!canvas) return null
      const rect = canvas.getBoundingClientRect()
      const scaleX = rect.width / canvas.width
      const scaleY = rect.height / canvas.height
      const domX = rect.left + spawn.x * scaleX
      const domY = rect.top + (spawn.y + spawn.width * 0.7 + 12) * scaleY

      return (
        <div
          style={{
            position: 'fixed',
            left: domX,
            top: domY,
            transform: 'translateX(-50%)',
            zIndex: 100,
            padding: '4px 10px',
            borderRadius: 999,
            background: 'rgba(0,0,0,0.7)',
            color: '#fff',
            fontWeight: 700,
            fontSize: 14,
            pointerEvents: 'none',
            opacity: exiting ? 0 : 1,
            transition: 'opacity 300ms ease-out',
            whiteSpace: 'nowrap',
          }}
        >
          {t('onboarding.tapHint.label')}
        </div>
      )
    }
    ```

    NB: `pointerEvents: 'none'` гарантирует что label не блокирует box-tap.
