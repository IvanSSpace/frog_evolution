    ```typescript
    import Phaser from 'phaser'

    const TEXTURE_KEY = 'onb-confetti-pixel'

    export interface ConfettiBurstOptions {
      scene: Phaser.Scene
      x: number
      y: number
      palette: number[]
      count?: number // default 40
      lifespanMs?: number // default 1200
      depth?: number // default 6000 (above tutorial rings)
    }

    /**
     * Ensure confetti texture exists в scene.
     * Generates 4x4 white pixel chunk (tint applied via emitter).
     */
    function ensureConfettiTexture(scene: Phaser.Scene): void {
      if (scene.textures.exists(TEXTURE_KEY)) return
      const gfx = scene.add.graphics({ x: 0, y: 0 })
      gfx.fillStyle(0xffffff, 1)
      gfx.fillRect(0, 0, 4, 4)
      gfx.generateTexture(TEXTURE_KEY, 4, 4)
      gfx.destroy()
    }

    /**
     * Phase 23 Beat 4: location unlock confetti burst.
     * Single-shot — auto-destroys after lifespanMs + buffer.
     */
    export class ConfettiBurst {
      static fire(opts: ConfettiBurstOptions): void {
        const scene = opts.scene
        ensureConfettiTexture(scene)

        const count = opts.count ?? 40
        const lifespan = opts.lifespanMs ?? 1200
        const depth = opts.depth ?? 6000

        // Phaser 3.60+ particle API:
        const emitter = scene.add.particles(opts.x, opts.y, TEXTURE_KEY, {
          speed: { min: 200, max: 450 },
          angle: { min: 200, max: 340 },
          gravityY: 700,
          lifespan,
          scale: { start: 1.5, end: 0.3 },
          alpha: { start: 1, end: 0 },
          tint: opts.palette,
          rotate: { min: 0, max: 360 },
          emitting: false,
        })
        emitter.setDepth(depth)
        emitter.explode(count, opts.x, opts.y)

        // Self-destruct after lifespan + buffer
        scene.time.delayedCall(lifespan + 300, () => {
          emitter.destroy()
        })
      }
    }
    ```

    **Important:**
    - НЕ tween'им frog.container.alpha — это standalone burst, не трогает frogs.
    - depth 6000 — выше TutorialPulseRing (5000), чтобы particles виднелись поверх rings
    - palette передаётся как массив tint hex чисел (Phaser автоматически random'ит per particle)

    ```tsx
    import { useEffect, useState } from 'react'
    import { useTranslation } from 'react-i18next'
    import { eventBus } from '../../store/eventBus'
    import { useOnboardingStore } from '../../store/onboarding/onboardingSlice'
    import './locationCelebration.css'

    // Локально mapped (mirror LOCATION_VISUAL + i18n key derivation).
    // Plan 23-06 может consolidate в shared util если другие места захотят reuse.
    const LOC_INFO: Record<number, { emoji: string; nameKey: string }> = {
      2: { emoji: '🌿', nameKey: 'locations.swamp' },
      3: { emoji: '🌲', nameKey: 'locations.forest' },
      6: { emoji: '✨', nameKey: 'locations.starMap' },
    }

    /**
     * Phase 23 Beat 4: DOM toast banner внизу.
     * Auto-fade 7s или tap-dismiss. One instance at a time
     * (если несколько unlock'ов подряд — queue?).
     *
     * Для demo-build акcept single concurrent toast — игрок не получит 2
     * unlock'а одновременно в нормальной игре.
     */
    export function LocationUnlockCelebration() {
      const { t } = useTranslation()
      const [active, setActive] = useState<number | null>(null)
      const [exiting, setExiting] = useState(false)

      useEffect(() => {
        const onStart = ({ locationId }: { locationId: number }) => {
          if (!LOC_INFO[locationId]) return // только known locations
          setActive(locationId)
          setExiting(false)
        }
        const onDismiss = () => {
          setExiting(true)
          window.setTimeout(() => setActive(null), 300)
        }
        eventBus.on('onboarding:locationCelebrationStart', onStart)
        eventBus.on('onboarding:locationCelebrationDismiss', onDismiss)
        return () => {
          eventBus.off('onboarding:locationCelebrationStart', onStart)
          eventBus.off('onboarding:locationCelebrationDismiss', onDismiss)
        }
      }, [])

      // Auto-fade 7s
      useEffect(() => {
        if (active === null || exiting) return
        const timer = window.setTimeout(() => {
          setExiting(true)
          window.setTimeout(() => setActive(null), 300)
        }, 7000)
        return () => clearTimeout(timer)
      }, [active, exiting])

      if (active === null) return null
      const info = LOC_INFO[active]
      if (!info) return null

      const name = t(info.nameKey)

      return (
        <div
          className={`onb-loc-toast${exiting ? ' is-exiting' : ''}`}
          role="status"
          aria-live="polite"
          onClick={() => {
            setExiting(true)
            window.setTimeout(() => setActive(null), 300)
          }}
          style={{
            position: 'fixed',
            bottom: 120, // above BottomBar
            left: '50%',
            zIndex: 101,
            padding: '12px 20px',
            borderRadius: 999,
            background: 'linear-gradient(180deg, #ec4899, #db2777)',
            color: '#fff',
            fontWeight: 800,
            fontSize: 15,
            boxShadow:
              'inset 0 1px 0 rgba(255,255,255,0.45), 0 6px 16px rgba(0,0,0,0.4)',
            cursor: 'pointer',
            touchAction: 'manipulation',
            maxWidth: 'calc(100vw - 32px)',
            whiteSpace: 'nowrap',
            overflow: 'hidden',
            textOverflow: 'ellipsis',
          }}
        >
          {t('onboarding.location.unlocked', { name: `${info.emoji} ${name}` })}
        </div>
      )
    }
    ```
