    ```typescript
    import Phaser from 'phaser'

    export interface GhostFrogTrailOptions {
      scene: Phaser.Scene
      // Source sprite — frog body texture key и tint
      textureKey: string
      tint?: number
      // Coords
      source: { x: number; y: number }
      target: { x: number; y: number }
      // Default 1200
      durationMs?: number
      // Default 0.5
      alpha?: number
      // Default 800ms pause между loops
      pauseMs?: number
      // Default 3 loops
      loops?: number
      // Default 5000
      depth?: number
      // Callback после finish всех loops
      onComplete?: () => void
    }

    /**
     * Phase 23 Beat 3: ghost-frog drag-trail animation.
     *
     * Spawn semi-transparent копия sprite'а. Tween position по smooth quadratic curve
     * source → target, fade-out + small burst при arrival. Loop N раз с pauses.
     *
     * Cancel via .destroy() — stops mid-animation, removes ghost.
     */
    export class GhostFrogTrail {
      private scene: Phaser.Scene
      private ghost: Phaser.GameObjects.Image | null = null
      private currentTween: Phaser.Tweens.Tween | null = null
      private pauseTimer: Phaser.Time.TimerEvent | null = null
      private destroyed = false
      private loopsRemaining: number
      private opts: GhostFrogTrailOptions

      constructor(opts: GhostFrogTrailOptions) {
        this.opts = opts
        this.scene = opts.scene
        this.loopsRemaining = opts.loops ?? 3
        this.runOneLoop()
      }

      private runOneLoop(): void {
        if (this.destroyed) return
        if (this.loopsRemaining <= 0) {
          this.opts.onComplete?.()
          return
        }

        const { textureKey, tint, source, target } = this.opts
        const durationMs = this.opts.durationMs ?? 1200
        const alpha = this.opts.alpha ?? 0.5
        const depth = this.opts.depth ?? 5000

        // Curve: arc up
        const midX = (source.x + target.x) / 2
        const midY = Math.min(source.y, target.y) - 50
        const curve = new Phaser.Curves.QuadraticBezier(
          new Phaser.Math.Vector2(source.x, source.y),
          new Phaser.Math.Vector2(midX, midY),
          new Phaser.Math.Vector2(target.x, target.y),
        )

        this.ghost = this.scene.add.image(source.x, source.y, textureKey)
        if (tint != null) this.ghost.setTint(tint)
        this.ghost.setAlpha(alpha)
        this.ghost.setDepth(depth)

        const pos = { t: 0 }
        this.currentTween = this.scene.tweens.add({
          targets: pos,
          t: 1,
          duration: durationMs,
          ease: 'Sine.easeInOut',
          onUpdate: () => {
            if (this.destroyed || !this.ghost) return
            const p = curve.getPoint(pos.t)
            this.ghost.x = p.x
            this.ghost.y = p.y
          },
          onComplete: () => {
            if (this.destroyed || !this.ghost) return
            // Small burst at arrival: scale + fade
            this.scene.tweens.add({
              targets: this.ghost,
              alpha: 0,
              scale: 1.3,
              duration: 300,
              ease: 'Sine.easeOut',
              onComplete: () => {
                this.ghost?.destroy()
                this.ghost = null
                this.loopsRemaining -= 1
                if (this.destroyed) return
                // Pause then loop
                this.pauseTimer = this.scene.time.delayedCall(
                  this.opts.pauseMs ?? 800,
                  () => {
                    this.runOneLoop()
                  },
                )
              },
            })
          },
        })
      }

      /**
       * Stop animation immediately, destroy ghost. Idempotent.
       */
      destroy(): void {
        if (this.destroyed) return
        this.destroyed = true

        this.currentTween?.stop()
        this.currentTween = null

        this.pauseTimer?.remove()
        this.pauseTimer = null

        if (this.ghost) {
          this.scene.tweens.add({
            targets: this.ghost,
            alpha: 0,
            duration: 200,
            onComplete: () => {
              this.ghost?.destroy()
              this.ghost = null
            },
          })
        }
      }
    }
    ```

    **Important — texture reuse:**
    `textureKey` — это Phaser texture key загруженной frog body (например `'frog-body-L1'`
    или whatever existing). BoxController/FrogSpawner знают это. Передадим из callsite.

    **Important — НЕ tween'им frog.container.alpha:**
    Ghost — отдельный sprite, не существующая frog. Чистый clone, никакой mutation
    реальной лягушки.

    DOM label «Перетащи одну на другую» позиционированный между двумя frogs.
    После merge показывает one-time toast «Готово! Дальше мерджи всё подряд» (3s auto).

    Структура:
    1. Subscribe на 'tutorial:mergeDemoStart' для anchor coords
    2. Subscribe на 'tutorial:firstMerge' для toast
    3. Auto-fade label через 8с (или dismiss on tutorial:firstMerge)
    4. Toast: render если firstMerge получен, auto-hide 3s

    ```tsx
    import { useEffect, useState } from 'react'
    import { useTranslation } from 'react-i18next'
    import { eventBus } from '../../store/eventBus'
    import { useOnboardingStore } from '../../store/onboarding/onboardingSlice'

    export function MergeHintOverlay() {
      const { t } = useTranslation()
      const [anchor, setAnchor] = useState<{ x: number; y: number } | null>(null)
      const [labelExiting, setLabelExiting] = useState(false)
      const [toastVisible, setToastVisible] = useState(false)

      useEffect(() => {
        const onStart = (p: {
          sourceX: number
          sourceY: number
          targetX: number
          targetY: number
        }) => {
          const midX = (p.sourceX + p.targetX) / 2
          // Place label below the lower of two
          const labelY = Math.max(p.sourceY, p.targetY) + 60
          setAnchor({ x: midX, y: labelY })
          setLabelExiting(false)
        }
        const onMerge = () => {
          setLabelExiting(true)
          window.setTimeout(() => setAnchor(null), 300)
          setToastVisible(true)
          window.setTimeout(() => setToastVisible(false), 3000)
        }
        eventBus.on('tutorial:mergeDemoStart', onStart)
        eventBus.on('tutorial:firstMerge', onMerge)
        return () => {
          eventBus.off('tutorial:mergeDemoStart', onStart)
          eventBus.off('tutorial:firstMerge', onMerge)
        }
      }, [])

      // Auto-fade 8с (Beat 3 чуть дольше)
      useEffect(() => {
        if (!anchor || labelExiting) return
        const timer = window.setTimeout(() => {
          setLabelExiting(true)
          window.setTimeout(() => {
            setAnchor(null)
            useOnboardingStore.getState().markFirstMergeSeen()
          }, 300)
        }, 8000)
        return () => clearTimeout(timer)
      }, [anchor, labelExiting])

      // Conversion game-coord → DOM (same pattern как TapHintOverlay)
      const toDom = (x: number, y: number) => {
        const canvas = document.querySelector('canvas')
        if (!canvas) return null
        const rect = canvas.getBoundingClientRect()
        const sx = rect.width / canvas.width
        const sy = rect.height / canvas.height
        return { x: rect.left + x * sx, y: rect.top + y * sy }
      }

      const labelDom = anchor ? toDom(anchor.x, anchor.y) : null

      return (
        <>
          {labelDom && (
            <div
              style={{
                position: 'fixed',
                left: labelDom.x,
                top: labelDom.y,
                transform: 'translateX(-50%)',
                zIndex: 100,
                padding: '4px 12px',
                borderRadius: 999,
                background: 'rgba(0,0,0,0.7)',
                color: '#fff',
                fontWeight: 700,
                fontSize: 14,
                pointerEvents: 'none',
                opacity: labelExiting ? 0 : 1,
                transition: 'opacity 300ms ease-out',
                whiteSpace: 'nowrap',
              }}
            >
              {t('onboarding.mergeHint.label')}
            </div>
          )}

          {toastVisible && (
            <div
              style={{
                position: 'fixed',
                bottom: 120, // выше BottomBar
                left: '50%',
                transform: 'translateX(-50%)',
                zIndex: 101,
                padding: '10px 20px',
                borderRadius: 999,
                background: 'linear-gradient(180deg, #f9a8d4, #ec4899)',
                color: '#fff',
                fontWeight: 800,
                fontSize: 14,
                boxShadow:
                  'inset 0 1px 0 rgba(255,255,255,0.45), 0 4px 12px rgba(0,0,0,0.35)',
                pointerEvents: 'none',
                animation: 'onb-toast-slide-up 300ms ease-out',
              }}
            >
              {t('onboarding.mergeHint.success')}
            </div>
          )}

          <style>{`
            @keyframes onb-toast-slide-up {
              from { opacity: 0; transform: translate(-50%, 20px); }
              to { opacity: 1; transform: translate(-50%, 0); }
            }
          `}</style>
        </>
      )
    }
    ```

    **`OnboardingController.tsx` — integration:**

    Controller отвечает за:
    1. Watch `firstBoxTapSeen=true && firstMergeSeen=false && l1FrogsCount >= 2`
    2. При trigger: создать 2 TutorialPulseRing + GhostFrogTrail + emit 'tutorial:mergeDemoStart'
    3. Player начал реальный drag → cancel demo (GhostFrogTrail.destroy + rings stay)
    4. Player начал и завершил merge → MergeController emit'нет 'tutorial:firstMerge' → dismiss rings

    Это сложнее чем Beat 2 — controller нуждается в Phaser scene access. Подход:
    создать вспомогательный helper class `MergeDemoCoordinator` который живёт в
    OnboardingController через useEffect ref:

    ```tsx
    import { useEffect, useState } from 'react'
    import { useGameStore } from '../../store/gameStore'
    import { useOnboardingStore } from '../../store/onboarding/onboardingSlice'
    import { eventBus } from '../../store/eventBus'
    import { TutorialPulseRing } from '../../game/effects/TutorialPulseRing'
    import { GhostFrogTrail } from '../../game/effects/GhostFrogTrail'
    import { WelcomeModal } from './WelcomeModal'
    import { TapHintOverlay } from './TapHintOverlay'
    import { MergeHintOverlay } from './MergeHintOverlay'

    export function OnboardingController() {
      const welcomeSeen = useOnboardingStore((s) => s.welcomeSeen)
      const firstBoxTapSeen = useOnboardingStore((s) => s.firstBoxTapSeen)
      const firstMergeSeen = useOnboardingStore((s) => s.firstMergeSeen)
      const locationFrogs = useGameStore((s) => s.locationFrogs)
      const currentLocation = useGameStore((s) => s.currentLocation)
      const carriers = useGameStore((s: any) => s.carriers ?? [])

      // Beat 3 demo coordinator — local state
      const [demoActive, setDemoActive] = useState(false)

      useEffect(() => {
        if (!welcomeSeen || !firstBoxTapSeen || firstMergeSeen) return
        if (demoActive) return

        // Count L1 carrier-free frogs on current location
        const frogs = (locationFrogs[currentLocation - 1] ?? []) as any[]
        const l1Free = frogs.filter(
          (f: any) =>
            f.level === 1 &&
            !carriers.some((c: any) => c.frogId === f.id),
        )
        if (l1Free.length < 2) return

        // Pick first two
        const src = l1Free[0]
        const tgt = l1Free[1]

        // Phaser scene access — через global game.scene or scene.registry
        // Simplest: scene находится в window.__phaserMainScene (set by MainScene constructor)
        // Если такого global нет — нужно дополнительный setup (см. Task 3 Phaser side).
        const scene = (window as any).__phaserMainScene as Phaser.Scene | undefined
        if (!scene) return

        setDemoActive(true)

        // Create 2 rings (frog-sized smaller radius)
        const ring1 = new TutorialPulseRing({
          scene,
          target: src.container ?? src,
          radius: 30,
        })
        const ring2 = new TutorialPulseRing({
          scene,
          target: tgt.container ?? tgt,
          radius: 30,
        })

        // Emit anchor for DOM label
        eventBus.emit('tutorial:mergeDemoStart', {
          sourceX: src.x ?? src.container?.x ?? 0,
          sourceY: src.y ?? src.container?.y ?? 0,
          targetX: tgt.x ?? tgt.container?.x ?? 0,
          targetY: tgt.y ?? tgt.container?.y ?? 0,
        })

        // Ghost trail loop
        const ghost = new GhostFrogTrail({
          scene,
          textureKey: src.bodyTextureKey ?? 'frog-body-L1', // verify actual key
          source: { x: src.x, y: src.y },
          target: { x: tgt.x, y: tgt.y },
          loops: 3,
        })

        // Cleanup на dismiss
        const onMerge = () => {
          ring1.destroy()
          ring2.destroy()
          ghost.destroy()
          setDemoActive(false)
        }
        eventBus.on('tutorial:firstMerge', onMerge)

        // Auto-fade 8с — same logic как MergeHintOverlay
        const timer = window.setTimeout(() => {
          ring1.destroy()
          ring2.destroy()
          ghost.destroy()
          useOnboardingStore.getState().markFirstMergeSeen()
          setDemoActive(false)
        }, 8000)

        // Cancel demo если player начал реальный drag (frog:pickup event)
        const onPickup = () => {
          ghost.destroy()
          // Rings stay — assist player visually
        }
        eventBus.on('frog:pickup', onPickup)

        return () => {
          clearTimeout(timer)
          eventBus.off('tutorial:firstMerge', onMerge)
          eventBus.off('frog:pickup', onPickup)
          ring1.destroy()
          ring2.destroy()
          ghost.destroy()
        }
      }, [
        welcomeSeen,
        firstBoxTapSeen,
        firstMergeSeen,
        locationFrogs,
        currentLocation,
        carriers,
        demoActive,
      ])

      return (
        <>
          {!welcomeSeen && <WelcomeModal />}
          {welcomeSeen && !firstBoxTapSeen && <TapHintOverlay />}
          {firstBoxTapSeen && !firstMergeSeen && <MergeHintOverlay />}
          {/* Plan 23-05: location celebration */}
        </>
      )
    }
    ```
